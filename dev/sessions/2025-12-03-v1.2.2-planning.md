```python
def execute_line(self, line: str, line_num: int):
    # Check debug mode
    if self.debug_engine.enabled:
        # Trace logging
        self.debug_engine.trace_line(line_num, line, self.variables)

        # Check breakpoints
        if self.debug_engine.should_pause(line_num):
            self.debug_engine.pause()
            # Wait for user input (STEP, CONTINUE, INSPECT)
            return PAUSED

    # Normal execution
    return self._execute_command(line)
```

#### Task 2: Gap Analysis Phase (200 lines)

**Workflow Implementation:**

```python
# Phase 1: Gap Analysis
ECHO "=== Phase 1: Gap Analysis ==="

# Load knowledge topics master list
LOAD knowledge_topics FROM core/data/knowledge_topics.json

# Scan existing knowledge directory
SCAN knowledge/ AS existing_guides

# Compare and identify gaps
FOR category IN ${knowledge_topics.categories}
    SET category_gaps []
    FOR topic IN ${category.topics}
        IF NOT EXISTS knowledge/${category}/${topic.id}.md
            APPEND category_gaps ${topic}
        END

    # Priority ranking (priority 1 > 2 > 3)
    SORT category_gaps BY priority ASC

    # Save gaps for this category
    SAVE category_gaps TO memory/workflows/state/knowledge-expansion/gaps_${category}.json
END

# Generate summary
SET total_gaps COUNT(gaps_*.json)
ECHO "Total gaps identified: ${total_gaps}"
```

#### Task 4: Quality Review Phase (250 lines)

**Workflow Implementation:**
```python
# Phase 3: Quality Review
ECHO "=== Phase 3: Quality Review ==="

# Load generated articles from Phase 2
LOAD batch FROM memory/workflows/state/knowledge-expansion/batch.json

SET approved []
SET rejected []
SET regenerated []

FOR article IN ${batch}
    # Review using DOCS REVIEW (4-dimension scoring)
    DOCS REVIEW memory/drafts/knowledge/${article.category}/${article.id}.md

    # Get review score
    SET review_score ${DOCS.LAST_REVIEW.score}
    SET review_dimensions ${DOCS.LAST_REVIEW.dimensions}

    # Quality threshold: 85%
    IF ${review_score} >= 85
        APPEND approved ${article}
        ECHO "✅ ${article.title}: ${review_score}%"
    ELSE
        # Attempt REGEN (one retry only)
        ECHO "⚠️  ${article.title}: ${review_score}% (below threshold)"
        ECHO "Attempting regeneration..."

        DOCS REGEN memory/drafts/knowledge/${article.category}/${article.id}.md
            --feedback "${review_dimensions.lowest_scoring}"

        # Re-review
        DOCS REVIEW memory/drafts/knowledge/${article.category}/${article.id}.md
        SET regen_score ${DOCS.LAST_REVIEW.score}

        IF ${regen_score} >= 85
            APPEND approved ${article}
            APPEND regenerated ${article}
            ECHO "✅ ${article.title}: ${regen_score}% (after REGEN)"
        ELSE
            APPEND rejected ${article}
            ECHO "❌ ${article.title}: ${regen_score}% (failed after REGEN)"
        END
    END
END

# Save review results
SAVE {
    "approved": ${approved},
    "rejected": ${rejected},
    "regenerated": ${regenerated},
    "approval_rate": PERCENTAGE(COUNT(approved) / COUNT(batch))
} TO memory/workflows/state/knowledge-expansion/review.json

ECHO "Review complete: ${COUNT(approved)}/${COUNT(batch)} approved (${approval_rate}%)"
```

#### Task 5: Validation & Commit Phase (200 lines)

**Workflow Implementation:**
```python
# Phase 4: Standards Validation
ECHO "=== Phase 4: Standards Validation ==="

LOAD approved FROM memory/workflows/state/knowledge-expansion/review.json

SET validated []
SET validation_errors []

FOR article IN ${approved}
    SET file memory/drafts/knowledge/${article.category}/${article.id}.md

    # Validate markdown structure
    VALIDATE markdown ${file} AS structure_valid

    # Validate metadata (frontmatter)
    VALIDATE metadata ${file} AS metadata_valid

    # Validate citations (at least 2 sources)
    VALIDATE citations ${file} MINIMUM 2 AS citations_valid

    # Validate word count (minimum 500 words)
    VALIDATE wordcount ${file} MINIMUM 500 AS wordcount_valid

    # All checks must pass
    IF ${structure_valid} AND ${metadata_valid} AND ${citations_valid} AND ${wordcount_valid}
        APPEND validated ${article}
    ELSE
        APPEND validation_errors {
            "article": ${article},
            "structure": ${structure_valid},
            "metadata": ${metadata_valid},
            "citations": ${citations_valid},
            "wordcount": ${wordcount_valid}
        }
    END
END

SAVE validated TO memory/workflows/state/knowledge-expansion/validation.json

# Phase 5: Smart Commit
ECHO "=== Phase 5: Smart Commit ==="

SET committed []
SET commit_errors []

FOR article IN ${validated}
    SET source memory/drafts/knowledge/${article.category}/${article.id}.md
    SET dest knowledge/${article.category}/${article.id}.md

    # Overwrite protection (backup if exists)
    IF EXISTS ${dest}
        BACKUP ${dest} TO memory/system/backup/knowledge/
    END

    # Copy to knowledge bank
    COPY ${source} TO ${dest}

    # Git add
    RUN git add ${dest}

    APPEND committed ${article}
END

# Git commit (batch)
IF COUNT(committed) > 0
    RUN git commit -m "knowledge: Add ${COUNT(committed)} new guides (automated workflow v1.2.2)"
    ECHO "✅ Committed ${COUNT(committed)} articles to knowledge bank"
ELSE
    ECHO "⚠️  No articles to commit"
END

SAVE committed TO memory/workflows/state/knowledge-expansion/commit.json
```
